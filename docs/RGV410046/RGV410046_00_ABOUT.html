Calendário de aulas

Clique aqui e veja como o calendário foi feito! 
library(calendR)
# tutorial calendR
# https://r-coder.com/calendar-plot-r/
events <- rep(NA, 30)
events[c(8, 10:12,
         15, 17:19)] <- "Aula"
# Creating the calendar
calendR(month = 9,
        weeknames = c("Seg", "Ter", "Qua", "Qui", "Sex", "Sáb", "Dom"),
        special.days = events,
        special.col =  c("lightgreen"),
        title = "RGV410046\nIntrodução à linguagem R de programação", 
        subtitle = "Cronograma de aulas 2025/02 (Setembro) - 08:20 às 11:50",
        mbg.col = "forestgreen",            
        months.col = "white",
        lunar = TRUE,
        lunar.col = "gold",
        day.size = 4,
        legend.pos = "bottom")
