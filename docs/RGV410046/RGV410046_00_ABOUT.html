Calendário de aulas

Clique aqui e veja como o calendário foi feito! 
library(calendR)
# tutorial calendR
# https://r-coder.com/calendar-plot-r/
events <- rep(NA, 30)
events[c(11, 14, 15, 18, 21, 22, 25, 28)] <- "08:20 às 11:50"
events[27] <- "Entrega do desafio (sem atividade presencial)"

# img <- "figs/calendr.png"

# Creating the calendar
calendR(year = 2022,
        start_date ="2023-09-01",
        end_date = "2023-09-30",
        special.days = events,
        special.col =  c("lightgreen", "salmon"),
        title = "RGV410046\nIntrodução à linguagem R de programação", 
        subtitle = "Cronograma de aulas",
        mbg.col = "forestgreen",            
        months.col = "white",
        # bg.img = img,
        legend.pos = "bottom")
