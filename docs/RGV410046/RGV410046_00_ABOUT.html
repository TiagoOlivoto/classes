Calendário de aulas

Clique aqui e veja como o calendário foi feito! 
library(calendR)
# tutorial calendR
# https://r-coder.com/calendar-plot-r/
events <- rep(NA, 61)
events[c(21,  28)] <- "08:20 às 11:50"
events[c(23:25, 29:32)] <- "13:30 às 16:00"

img <- "figs/calendr.png"

# Creating the calendar
calendR(year = 2022,
        start_date ="2022-11-01",
        end_date = "2022-12-31",
        special.days = events,
        special.col = c("lightsalmon", "lightgreen"),
        title = "RGV410046 - Introdução à linguagem R de programação", 
        subtitle = "Cronograma de aulas",
        mbg.col = "forestgreen",            
        months.col = "white",
        bg.img = img,
        legend.pos = "bottom")
